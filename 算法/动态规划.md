### 斐波那契数列
```
// 普通递归
const fib = (n) => {
    if (n <= 2) {
        return 1;
    }
    return fib(n - 1) + fib(n - 2)
}

console.log(fib(6))
// 备忘录
const fibonacci = function() {
    let memo = [0, 1];
    let fib = function(n) {
        if (memo[n] === undefined) {
            memo[n] = fib(n-1) + fib(n-2) // 状态转移方程
        }
        return memo[n];
    }
    return fib;
}()

console.log(fibonacci(30));
// for循环
var fibonacci = function (n) {
    let n1 = 1; n2 = 1;
    for (let i = 2; i < n; i++) {
        [n1, n2] = [n2, n1 + n2]
    }
    return n2
}
console.log(fibonacci(30))
```
