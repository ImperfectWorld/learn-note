## 快排

```
const quickSort = (arr) => {
    if (arr.length === 0) {
        return [];
    }
    let nodeIndex = Math.floor(arr.length / 2);
    let node = arr.splice(nodeIndex, 1)[0];
    let left = [];
    let right = [];

    for (let i = 0; i < arr.length; i++) {
        if (arr[i] < node) {
            left.push(arr[i]);
        }
        else {
            right.push(arr[i])
        }
    }
    return quickSort(left).concat(node, quickSort(right));
}
```
